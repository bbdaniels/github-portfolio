Usage

<!-- 1. Include the script --> <script src="https://yourusername.github.io/GitGlue/GitHubPinner.js"></script> <!-- 2. Add a profile card --> <div data="https://github.com/username" class="github-pinner" style="visibility: hidden;"></div> <!-- 3. Add a repository card --> <div data="https://github.com/username/repo" class="github-pinner" style="visibility: hidden;"></div> <!-- 4. Show all repositories with search --> <div data="https://github.com/username?tab=repositories" class="github-pinner" style="visibility: hidden;"></div> <!-- Add "flat" class for alternative style --> <div data="https://github.com/username" class="github-pinner flat" style="visibility: hidden;"></div>
